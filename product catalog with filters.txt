    <div>
      <label for="categoryFilter">Category:</label>
      <select id="categoryFilter">
        <option value="all">All</option>
        <option value="electronics">Electronics</option>
        <option value="clothing">Clothing</option>
        <option value="books">Books</option>
      </select>

      <label for="minPrice">Min Price:</label>
      <input type="number" id="minPrice" placeholder="0" />

    </div>

    <!-- Products will be rendered here -->
  </div>

  <script>
    const products = [
      { id: 1, name: "Smartphone", category: "electronics", price: 499 },
      { id: 2, name: "Laptop", category: "electronics", price: 899 },
      { id: 3, name: "T-shirt", category: "clothing", price: 19 },
      { id: 4, name: "Jeans", category: "clothing", price: 49 },
      { id: 5, name: "Novel", category: "books", price: 15 },
      { id: 6, name: "Headphones", category: "electronics", price: 79 },
      { id: 7, name: "Jacket", category: "clothing", price: 89 },
      { id: 8, name: "Cookbook", category: "books", price: 25 },
    ];

    const categoryFilter = document.getElementById("categoryFilter");
    const minPrice = document.getElementById("minPrice");
    const maxPrice = document.getElementById("maxPrice");
    const productGrid = document.getElementById("productGrid");

    function renderProducts(filteredProducts) {
      productGrid.innerHTML = "";

      if (filteredProducts.length === 0) {
        productGrid.innerHTML = "<p>No products found.</p>";
        return;
      }

      filteredProducts.forEach(product => {
        const div = document.createElement("div");
        div.className = "product";
        div.innerHTML = `
          <h3>${product.name}</h3>
          <p>Category: ${product.category}</p>
          <p>Price: $${product.price}</p>
        `;
        productGrid.appendChild(div);
      });
    }

    function filterProducts() {
      const category = categoryFilter.value;
      const min = parseFloat(minPrice.value) || 0;
      const max = parseFloat(maxPrice.value) || Infinity;

      const filtered = products.filter(p => {
        const inCategory = category === "all" || p.category === category;
        const inPriceRange = p.price >= min && p.price <= max;
        return inCategory && inPriceRange;
      });

      renderProducts(filtered);
    }

    // Event Listeners
    categoryFilter.addEventListener("change", filterProducts);
    minPrice.addEventListener("input", filterProducts);
    maxPrice.addEventListener("input", filterProducts);

    // Initial render
    renderProducts(products);
  </script>
